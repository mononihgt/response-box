## 1. 反应时参数说明

### 1.1 基本参数

反应盒返回的数据帧包含以下信息：

| 字节位置 | 内容 | 说明 |
|---------|------|------|
| 0 | 帧头 (0xFB) | 固定值，用于验证数据 |
| 1 | 按键编号 (1-5) | 1=绿, 2=红, 3=蓝, 4=黄, 5=白 |
| 2-3 | t1 (大端16位) | **刺激到按键的时间（反应时）** |
| 4-5 | t2 (大端16位) | **按键按下到释放的时间** |
| 6-7 | 校验字节 | 用于验证数据完整性 |

### 1.2 核心参数定义

#### t1 - 反应时 (Reaction Time)
- **定义**: 从刺激呈现（触发计时命令 `0xFB 0x00`）到被试按下按键的时间
- **单位**: 毫秒 (ms)
- **范围**: 0 - 65535 ms (16位无符号整数)
- **精度**: 1 ms
- **用途**: 这是心理学实验中最重要的测量值

#### t2 - 按键持续时间 (Key Press Duration)
- **定义**: 按键从按下到释放的时间
- **单位**: 毫秒 (ms)
- **用途**: 
  - 检测按键质量（过短可能是误触，过长可能是犹豫）
  - 分析按键力度和决策确定性
  - 排除异常试次

## 2. 数据提取代码

### 2.1 基础提取

```python
import serial
import struct

def extract_reaction_data(port='COM3', baud=115200):
    """
    提取反应盒数据的完整示例
    
    返回:
        dict: 包含所有参数的字典
    """
    with serial.Serial(port, baud, timeout=10) as ser:
        # 1. 清空缓冲区
        ser.reset_input_buffer()
        
        # 2. 停止之前的计时
        ser.write(b'\xFC\x00')
        
        # 3. 触发计时（在刺激呈现时调用）
        trigger_time = time.perf_counter()  # 记录触发时间
        ser.write(b'\xFB\x00')
        
        # 4. 等待按键响应
        frame = ser.read(8)
        receive_time = time.perf_counter()  # 记录接收时间
        
        # 5. 验证数据
        if len(frame) != 8 or frame[0] != 0xFB:
            raise ValueError("数据帧格式错误")
        
        # 6. 解析参数
        key = frame[1]  # 按键编号
        t1, t2 = struct.unpack('>HH', frame[2:6])  # 大端解析
        checksum = int.from_bytes(frame[6:], byteorder='big')
        
        # 7. 计算额外参数
        python_delay = (receive_time - trigger_time) * 1000  # Python处理延迟
        
        return {
            'key': key,
            'key_name': decode_key(key),
            'reaction_time_ms': t1,
            'reaction_time_sec': t1 / 1000.0,
            'press_duration_ms': t2,
            'press_duration_sec': t2 / 1000.0,
            'checksum': checksum,
            'python_delay_ms': python_delay,
            'timestamp': time.time()
        }

def decode_key(key_num):
    """按键编号转名称"""
    keys = {1: 'Green', 2: 'Red', 3: 'Blue', 4: 'Yellow', 5: 'White'}
    return keys.get(key_num, 'Unknown')
```

### 2.2 批量数据收集

```python
import pandas as pd

def collect_reaction_data(n_trials=10):
    """
    收集多个试次的反应时数据
    
    参数:
        n_trials: 试次数量
    
    返回:
        DataFrame: 所有试次的数据
    """
    data_list = []
    
    for trial in range(1, n_trials + 1):
        print(f"\n试次 {trial}/{n_trials}")
        print("请按键...")
        
        try:
            # 获取一次反应
            data = extract_reaction_data()
            data['trial'] = trial
            data_list.append(data)
            
            # 显示结果
            print(f"  按键: {data['key_name']}")
            print(f"  反应时: {data['reaction_time_ms']} ms")
            
        except Exception as e:
            print(f"  错误: {e}")
            continue
    
    # 转换为DataFrame
    df = pd.DataFrame(data_list)
    return df

# 使用示例
df = collect_reaction_data(n_trials=20)
print(df.describe())
df.to_csv('reaction_data.csv', index=False)
```

## 3. 误差分析

### 3.1 误差来源

#### 系统误差（可测量、可校正）

| 误差类型 | 大小 | 来源 | 处理方法 |
|---------|------|------|---------|
| 串口通信延迟 | 0.5-2 ms | 数据传输 | 可通过校准补偿 |
| 反应盒硬件延迟 | < 1 ms | 芯片处理 | 固定值，可忽略 |
| 按键机械延迟 | < 0.5 ms | 物理接触 | 固定值，可忽略 |

#### 测量误差（随机、不可消除）

| 误差类型 | 大小 | 来源 | 处理方法 |
|---------|------|------|---------|
| 时钟分辨率 | ± 0.5 ms | 1ms精度 | 报告为±0.5ms |
| 按键抖动 | ± 1-2 ms | 机械抖动 | 统计方法处理 |
| 被试反应波动 | ± 50-100 ms | 个体差异 | 多次测量取平均 |

### 3.2 校准方法

#### 零延迟测试（需要硬件短接）

```python
def calibrate_system_delay(n_samples=100):
    """
    测量系统固有延迟
    
    方法：将触发信号和按键信号短接，模拟零延迟
    
    返回:
        dict: 校准结果
    """
    delays = []
    
    for i in range(n_samples):
        with serial.Serial(PORT, BAUD, timeout=1) as ser:
            # 清空缓冲
            ser.reset_input_buffer()
            ser.write(b'\xFC\x00')
            
            # 触发并计时
            start = time.perf_counter()
            ser.write(b'\xFB\x00')
            frame = ser.read(8)
            end = time.perf_counter()
            
            # 提取硬件报告的时间
            if len(frame) == 8 and frame[0] == 0xFB:
                t1 = struct.unpack('>H', frame[2:4])[0]
                python_time = (end - start) * 1000
                
                delays.append({
                    'hardware_delay': t1,
                    'python_delay': python_time,
                    'total_delay': t1 + python_time
                })
    
    # 统计分析
    df = pd.DataFrame(delays)
    
    result = {
        'hardware_delay_mean': df['hardware_delay'].mean(),
        'hardware_delay_std': df['hardware_delay'].std(),
        'python_delay_mean': df['python_delay'].mean(),
        'python_delay_std': df['python_delay'].std(),
        'total_delay_mean': df['total_delay'].mean(),
        'total_delay_std': df['total_delay'].std()
    }
    
    return result

# 运行校准
calibration = calibrate_system_delay()
print(f"系统延迟: {calibration['total_delay_mean']:.2f} ± {calibration['total_delay_std']:.2f} ms")
```

```python
def validate_with_known_delay(delay_ms=500, n_samples=20):
    """
    使用已知延迟验证测量精度
    
    参数:
        delay_ms: 设定的延迟时间
        n_samples: 测试次数
    """
    measured_delays = []
    
    for i in range(n_samples):
        with serial.Serial(PORT, BAUD, timeout=10) as ser:
            ser.reset_input_buffer()
            ser.write(b'\xFC\x00')
            ser.write(b'\xFB\x00')
            
            # 等待设定的延迟
            time.sleep(delay_ms / 1000.0)
            
            # 模拟按键（需要外部设备）
            # 或者手动按键
            
            frame = ser.read(8)
            if len(frame) == 8:
                t1 = struct.unpack('>H', frame[2:4])[0]
                measured_delays.append(t1)
    
    # 计算误差
    mean_measured = np.mean(measured_delays)
    error = mean_measured - delay_ms
    std_measured = np.std(measured_delays)
```

### 3.3 数据清洗

```python
def clean_reaction_data(df, min_rt=100, max_rt=2000, max_press=500):
    """
    清洗反应时数据
    
    参数:
        df: 原始数据
        min_rt: 最小有效反应时（ms）
        max_rt: 最大有效反应时（ms）
        max_press: 最大按压时长（ms）
    
    返回:
        DataFrame: 清洗后的数据
    """
    df_clean = df.copy()
    
    # 标记异常值
    df_clean['is_valid'] = (
        (df_clean['reaction_time_ms'] >= min_rt) &
        (df_clean['reaction_time_ms'] <= max_rt) &
        (df_clean['press_duration_ms'] <= max_press) &
        (df_clean['key'].isin([1, 2, 3, 4, 5]))
    )
    
    # 统计
    n_total = len(df_clean)
    n_valid = df_clean['is_valid'].sum()
    n_removed = n_total - n_valid
    
    print(f"总试次: {n_total}")
    print(f"有效试次: {n_valid} ({n_valid/n_total*100:.1f}%)")
    print(f"移除试次: {n_removed} ({n_removed/n_total*100:.1f}%)")
    
    # 移除异常原因分析
    if n_removed > 0:
        print("\n移除原因：")
        too_fast = ((df_clean['reaction_time_ms'] < min_rt) & ~df_clean['is_valid']).sum()
        too_slow = ((df_clean['reaction_time_ms'] > max_rt) & ~df_clean['is_valid']).sum()
        long_press = ((df_clean['press_duration_ms'] > max_press) & ~df_clean['is_valid']).sum()
        
        if too_fast > 0:
            print(f"  - 反应过快 (< {min_rt}ms): {too_fast}")
        if too_slow > 0:
            print(f"  - 反应过慢 (> {max_rt}ms): {too_slow}")
        if long_press > 0:
            print(f"  - 按键过长 (> {max_press}ms): {long_press}")
    
    return df_clean[df_clean['is_valid']]
```

### 4.1 基本统计

```python
def analyze_reaction_times(df):
    """
    反应时基本统计分析
    """
    rt_ms = df['reaction_time_ms']
    
    stats = {
        'n': len(rt_ms),
        'mean': rt_ms.mean(),
        'std': rt_ms.std(),
        'sem': rt_ms.sem(),
        'median': rt_ms.median(),
        'min': rt_ms.min(),
        'max': rt_ms.max(),
        'q25': rt_ms.quantile(0.25),
        'q75': rt_ms.quantile(0.75),
        'cv': rt_ms.std() / rt_ms.mean()  # 变异系数
    }
    
    print("反应时统计（毫秒）：")
    print(f"  样本量: {stats['n']}")
    print(f"  平均值: {stats['mean']:.2f} ± {stats['sem']:.2f} (SEM)")
    print(f"  标准差: {stats['std']:.2f}")
    print(f"  中位数: {stats['median']:.2f}")
    print(f"  范围: [{stats['min']}, {stats['max']}]")
    print(f"  四分位距: [{stats['q25']:.2f}, {stats['q75']:.2f}]")
    print(f"  变异系数: {stats['cv']:.3f}")
    
    return stats
```

### 4.2 条件比较

```python
def compare_conditions(df, condition_col='condition'):
    """
    比较不同条件下的反应时
    """
    import scipy.stats as stats
    
    conditions = df[condition_col].unique()
    
    print(f"\n条件比较（按 {condition_col}）：")
    print("=" * 60)
    
    # 各条件统计
    for cond in conditions:
        cond_data = df[df[condition_col] == cond]['reaction_time_ms']
        print(f"\n{cond}:")
        print(f"  N = {len(cond_data)}")
        print(f"  M = {cond_data.mean():.2f} ms")
        print(f"  SD = {cond_data.std():.2f} ms")
    
    # 两两比较（如果有2个条件）
    if len(conditions) == 2:
        group1 = df[df[condition_col] == conditions[0]]['reaction_time_ms']
        group2 = df[df[condition_col] == conditions[1]]['reaction_time_ms']
        
        # t检验
        t_stat, p_value = stats.ttest_ind(group1, group2)
        
        print(f"\nt检验结果：")
        print(f"  t({len(group1)+len(group2)-2}) = {t_stat:.3f}")
        print(f"  p = {p_value:.4f}")
        print(f"  Cohen's d = {(group1.mean() - group2.mean()) / np.sqrt((group1.std()**2 + group2.std()**2) / 2):.3f}")
```

```python
import matplotlib.pyplot as plt
import seaborn as sns

def plot_reaction_times(df):

    可视化反应时数据
```
